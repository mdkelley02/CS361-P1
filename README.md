<h3>Build</h3>
<code>
<pre>
javac fa/*.java fa/dfa/*.java 
</pre>
</code>
<h3>Run</h3>
<code>
<pre>
java fa.dfa.DFADriver path_to_test 
</pre>
</code>
<h3>Test</h3>
<code>
<pre>
java fa.dfa.DFADriverTester 
</pre>
</code>
<h3>Clean</h3>
<code>
<pre>
rm -rf fa/*.class && rm -rf fa/dfa/*.class 
</pre>
</code>
